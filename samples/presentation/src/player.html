HTML5网站中的视频播放器的使用

讲稿制作：Christopher

我们常用的HTML5视频播放器有：

  • mediaelementjs
  • videojs
  • 我们自己的播放器

为什么自己写一个播放器？

  • 提供更丰富的编程API
  • 更灵活的和可定制的样式
  • 遇到问题能更快速的解决

如何创建

define(['crown', 'crown.site/Scene'], function (crown, Scene) {
    var Scheme = crown.inherit("scene.BasicScene", Scene), proto = Scheme.prototype;
    proto.load = function ($content) {
        var self = this, deferred = $.Deferred();
        Scheme.uber.load.apply(this, arguments).done(function () {
            self._allLoaded = true;
            deferred.resolve();
        });
        stage.getSpinner().commit();
        return deferred.promise();
    };
    proto.enter = function (paper) {
        return Scheme.uber.enter.apply(this, arguments);
    };
    return Scheme;
});
